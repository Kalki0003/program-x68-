a =imread('D:\TYCS_509\sunder.jpg');
a1 = double(a);
a2 = rgb2gray(uint8(a1));
[row, col] = size(a2);
c = row * col;
h = zeros(1, 256);
z = zeros(1, 256);

for m = 1:row
    for n = 1:col
        t = a2(m, n);
        h(t + 1) = h(t + 1) + 1;
    end
end

pdf = h / c;
cdf = zeros(1, 256);
cdf(1) = pdf(1);

for x = 2:256
    cdf(x) = pdf(x) + cdf(x - 1);
end

new = round(cdf * 255);
b = zeros(row, col);

for p = 1:row
    for q = 1:col
        temp = a2(p, q) + 1;
        b(p, q) = new(temp);
        t = b(p, q);
        z(t + 1) = z(t + 1) + 1;
    end
end

subplot(2,2,1);
imshow(uint8(a2));
title('Original Image');

subplot(2,2,2);
bar(h);
title('Histogram of Original Image');

subplot(2,2,3);
imshow(uint8(b));
title('Histogram Equalized Image');

subplot(2,2,4);
bar(z);
title('Equalized Histogram');
